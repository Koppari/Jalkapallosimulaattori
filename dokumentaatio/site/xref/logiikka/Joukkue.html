11      public Joukkue() {
12      }
13  
14      //lisää maalivahdin, 4 puolustajaa, 4 keskikenttää ja 2 hyökkääjää
15      public void lisaaPelaajat() {
16          pelaajat.add(new Pelaaja(0));
17          for (int i = 1; i <= 4; i++) {
18              pelaajat.add(new Pelaaja(1));
19          }
20          for (int i = 1; i <= 4; i++) {
21              pelaajat.add(new Pelaaja(2));
22          }
23          for (int i = 1; i <= 2; i++) {
24              pelaajat.add(new Pelaaja(3));
25          }
26      }
27  
28      public void poistaPelaajat() {
29          pelaajat.clear();
30      }
31  
32      public void printPelaajat() {
33          for (Pelaaja p : this.pelaajat) {
34              System.out.println(p.toString());
35              System.out.println("");
36          }
37          if (this.pelaajat == null) {
38              System.out.println("Joukkueessa ei ole pelaajia!");
39          }
40      }
41  
42      public int kokonaisAttribuutit() {
43          int joukkueenVoima = 0;
44  
45          for (Pelaaja p : this.pelaajat) {
46              joukkueenVoima = joukkueenVoima + p.getKokonaisAttribuutit();
47          }
48  
49          if (joukkueenVoima < 0) {
50              joukkueenVoima = 0;
51          }
52  
53          this.joukkueenVoima = joukkueenVoima;
54          return joukkueenVoima;
55      }
56

59      }
