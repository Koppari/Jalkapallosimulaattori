3   import java.util.Random;
4   
5   public class RandomEventGeneraattori {
6   
7       // eri random eventtejä peliin
8       private static Random r = new Random();
9   
10      public RandomEventGeneraattori() {
11      }
12  
13      public static boolean maali(Joukkue j, Joukkue k) throws Exception {
14          if (j.joukkueenVoima == 0 || k.joukkueenVoima == 0) {
15              throw new Exception("Yhden tai molempien joukkueiden voima on 0!");
16          }
17  
18          //jos joukkueet samantasoisia, lähtökoht. >0.5 = j maali ja <0.5 = k maali
19          double mahdollisuusMaaliin = r.nextDouble();
20          double joukkueenParemmuus = 0;
21  
22          //antaa joukkueen j mahdollisuuden tehdä maalin
23          if (k.kokonaisAttribuutit() < j.kokonaisAttribuutit() || j.kokonaisAttribuutit() < k.kokonaisAttribuutit()) {
24              joukkueenParemmuus = 0.01 * (50 + (100 - (((double) k.joukkueenVoima / (double) j.joukkueenVoima) * 100)));
25          } else {
26              joukkueenParemmuus = 0.5;
27          }
28  
29          //palauttaa true jos maali on joukkueen j
30          if (mahdollisuusMaaliin < joukkueenParemmuus) {
31              return true;
32          } else {
33              return false;
34          }
35  
36      }
37  
38      public static Pelaaja maalinTekija(Joukkue j) {
39          //valitsee maalille tekijän, mitä hyökkäävämpi pelaaja sitä varmemmin maali häneltä
40          if (j.pelaajat.isEmpty()) {
41              return null;
42          }
43          return j.pelaajat.get(r.nextInt(j.pelaajat.size()));
44      }
45  
46  }
